public class IntegerGenerator {

    /**
     * Menghasilkan daftar bilangan bulat dari nilai awal hingga nilai akhir.
     *
     * @param start Nilai awal dari bilangan bulat yang akan dihasilkan.
     *              Untuk kasus ini, biasanya diatur ke 1.
     * @param end   Nilai akhir dari bilangan bulat yang akan dihasilkan.
     *              Untuk kasus ini, biasanya diatur ke 100.
     * @return List berisi bilangan bulat dari start hingga end (inklusif).
     * @throws IllegalArgumentException jika start lebih besar dari end atau jika start atau end berada di luar rentang 1 hingga 100.
     */
    public static List generateIntegers(int start, int end) {
        // Validasi input: Pastikan start <= end dan keduanya dalam rentang 1-100
        if (start > end || start < 1 || end > 100) {
            throw new IllegalArgumentException("Nilai start harus <= end dan keduanya berada dalam rentang 1 hingga 100.");
        }

        // Inisialisasi daftar untuk menyimpan bilangan bulat
        List integers = new ArrayList<>();

        // Loop dari start ke end untuk menambahkan bilangan bulat ke daftar
        for (int i = start; i <= end; i++) {
            integers.add(i);
        }

        return integers;
    }

    /**
     * Fungsi utama untuk menguji fungsi generateIntegers.
     *
     * @param args Argumen baris perintah (tidak digunakan dalam fungsi ini).
     */
    public static void main(String[] args) {
        try {
            // Menghasilkan dan menampilkan daftar bilangan bulat dari 1 hingga 100
            List integers = generateIntegers(1, 100);
            System.out.println("Bilangan bulat dari 1 hingga 100: " + integers);
        } catch (IllegalArgumentException e) {
            System.out.println("Kesalahan: " + e.getMessage());
        }
    }
}
